# curry

函数式编程中, 把参数降成单个的方式称为柯里化(currying)

``` js
    curry = (fn, a) => (b) => fn.call(this, b, a);
    const tailFactorial = curry(factorial, 1)
    tailFactorial(5) //120
    //一般转换
    function add(a, b) {
      return a + b
    }

    function add(a) {
      return function(b) {
        return a + b
      }
    }
    // use lamada function
    const add = a => b => a + b
```

## 例题

``` JS
// JavaScript
// 实现 curry 使得下面函数可以运行
var fn = curry(function(a, b, c) {
  console.log([a, b, c]);
});

fn("a", "b", "c") // ["a", "b", "c"]
fn("a", "b")("c") // ["a", "b", "c"]
fn("a")("b")("c") // ["a", "b", "c"]
fn("a")("b", "c") // ["a", "b", "c"]
```

这个有多个解题方法, 

``` JS
// JavaScript
// 这是一个辅助函数,他可以做到
// fn("a", "b", "c") // ["a", "b", "c"]
// 但问题是没有办法接受其他参数的形式,因为它在第一次运行以后就必须要接收到所有的参数
// 所以需要写一个curry函数来包装它,在返回足够的函数参数之后,将照常返回,否则,要再返回一个函数,来接收更多的参数
function sub_curry(fn) {
  var args = [].slice.call(arguments, 1);
  console.log('形成函数输出': ...args)
  return function() {
    console.log('执行函数输出': ...arguments)
    return fn.apply(this, args.concat([].slice.call(arguments)));
  };
}
// 这个就是用来包裹的curry函数
function curry(fn, length) {
  length = length || fn.length;
  var slice = Array.prototype.slice;
  return function() {
    // 做出判断
    if (arguments.length < length) {
      // combine返回的是符合curry的参数,即 fn,...args
      var combined = [fn].concat(slice.call(arguments));
      // 调用之后返回了一个function(){return fn.apply()}
      return curry(sub_curry.apply(this, combined), length - arguments.length);
    } else {
      // 返回调用的结果
      return fn.apply(this, arguments);
    }
  };
}
// 猜测 '形成函数输出'与 '生成函数输出'
// 结果为
// 形成函数输出 a
// 形成函数输出 b
// 生成函数输出 c
// 生成函数输出 b c

// 以fn(a)(b)(c)举例
// 第一次传入a
function sub_curry(fn) {
  var args = [].slice.call(arguments, 1);
  console.log('形成函数输出': ...args)
  // a
  return function() {
    console.log('执行函数输出': ...arguments)
    return fn.apply(this, args.concat([].slice.call(arguments)));
    // 等同于 fn.apply(this,[...args,...rest])
  };
}
// sub_curry返回
function untitled(...rest) {
  console.log('执行函数输出': ...arguments)
  // 这里形成一个闭包,args是['a']
  return fn.apply(this, [...args, ...rest])
}
// 然后将这个函数传入curry
// 在传入b后

function sub_curry(fn) {
  var args = [].slice.call(arguments, 1);
  console.log('形成函数输出': ...args)
  // b
  return function() {
    console.log('执行函数输出': ...arguments)
    return fn.apply(this, args.concat([].slice.call(arguments)));
  };
}
// sub_curry返回
function _untitled(..._rest) {

  console.log('执行函数输出': ...arguments)

  // 为了标记不同将b传入时候的fn和args使用下划线标记
  _fn = function untitled(...rest) {
    console.log('执行函数输出': ...arguments)
    // 这里形成一个闭包,args是['a']
    return fn.apply(this, [...args, ...rest])
  }
  // 这里形成一个闭包,_args是['b']
  return _fn.apply(this, [.._args, ..._rest])
}
// 然后在第三次传入c之后
// 将上面这个函数作为fn传入
// 这时候走到fn.apply(this, arguments);
// 那么这个时候相当于
function _untitled(..._rest) {
  // 为了标记不同将b传入时候的fn和args使用下划线标记
  console.log('执行函数输出': ...arguments)
  // 输出c
  _fn = function untitled(fn, ...rest) {
    console.log('执行函数输出': ...arguments)
    // 输出 b,c
    // 这个args是'a'
    return fn.apply(this, [...args, ...rest])
    // 相当于fn.apply(this,[a,b,c])
  }
  return _fn.apply(this, [..._args, ..._rest])
  // 这个_rest就是[c], _args是b
}('c')

// 终于以c作为参数调用函数了
// 然后执行_fn.apply(this, [_args, c])
// 然后再执行fn.apply(this,[args,...rest])
```

这段代码的思路是, 在 包装 `sub_curry` 的基础上改造, 在参数不够的时候, 将构建一个闭包, 返回一个引用了闭包内参数的函数, 此时如果依然不够, 那么将给闭包返回的函数再投入进来, 再构建一个闭包, 返回一个引用了闭包内参数的函数, 一旦参数个数满足, 就将那个返回的函数执行, 它会去每个闭包里获取存放的参数, 最后通过apply执行.
